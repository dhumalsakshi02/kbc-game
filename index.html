# Writing the Python GUI code for KBC Game with Tkinter kbc_code = """ import tkinter as tk from tkinter import messagebox import random # Questions, options, and correct answers questions = [ ("What is the capital of India?", ["Mumbai", "Chennai", "Delhi", "Kolkata"], "Delhi"), ("Which planet is known as the Red Planet?", ["Earth", "Mars", "Jupiter", "Saturn"], "Mars"), ("Who wrote the Indian National Anthem?", ["Tagore", "Gandhi", "Nehru", "Patel"], "Tagore"), ("What is the currency of Japan?", ["Yuan", "Won", "Yen", "Dollar"], "Yen"), ("Who painted the Mona Lisa?", ["Van Gogh", "Da Vinci", "Picasso", "Rembrandt"], "Da Vinci") ("Who is known as the Father of the Nation in India?", ["Jawaharlal Nehru", "Mahatma Gandhi", "Sardar Patel", "Subhas Bose"], "Mahatma Gandhi"), ("Which is the largest planet in our solar system?", ["Mars", "Saturn", "Jupiter", "Earth"], "Jupiter"), ("What is the national currency of Japan?", ["Yuan", "Won", "Yen", "Ringgit"], "Yen"), ("Which language is primarily used for Android app development?", ["Python", "Java", "Swift", "C++"], "Java"), ("How many players are there in a cricket team?", ["9", "10", "11", "12"], "11"), ("Who wrote the play 'Romeo and Juliet'?", ["Charles Dickens", "William Shakespeare", "Leo Tolstoy", "George Orwell"], "William Shakespeare"), ("Which organ is responsible for pumping blood?", ["Liver", "Brain", "Heart", "Lungs"], "Heart"), ("What gas do plants absorb from the atmosphere?", ["Oxygen", "Carbon Dioxide", "Nitrogen", "Hydrogen"], "Carbon Dioxide"), ("Who discovered gravity?", ["Albert Einstein", "Isaac Newton", "Galileo", "Thomas Edison"], "Isaac Newton"), ("What is the boiling point of water?", ["90°C", "80°C", "100°C", "120°C"], "100°C") ] # Shuffle questions random.shuffle(questions) class KBCGame: def __init__(self, root): self.root = root self.root.title("KBC Game") self.q_index = 0 self.score = 0 self.used_5050 = False self.used_skip = False self.used_poll = False self.question_label = tk.Label(root, text="", font=("Arial", 14), wraplength=500, justify="center") self.question_label.pack(pady=20) self.options = [] for i in range(4): btn = tk.Button(root, text="", width=40, font=("Arial", 12), command=lambda i=i: self.check_answer(i)) btn.pack(pady=5) self.options.append(btn) self.lifeline_frame = tk.Frame(root) self.lifeline_frame.pack(pady=10) self.btn_5050 = tk.Button(self.lifeline_frame, text="50:50", command=self.use_5050) self.btn_5050.grid(row=0, column=0, padx=10) self.btn_poll = tk.Button(self.lifeline_frame, text="Audience Poll", command=self.use_poll) self.btn_poll.grid(row=0, column=1, padx=10) self.btn_skip = tk.Button(self.lifeline_frame, text="Skip", command=self.use_skip) self.btn_skip.grid(row=0, column=2, padx=10) self.load_question() def load_question(self): if self.q_index >= len(questions): messagebox.showinfo("Game Over", f"Your final score is: {self.score}") self.root.quit() return self.current_question, self.current_options, self.correct_answer = questions[self.q_index] self.question_label.config(text=f"Q{self.q_index+1}: {self.current_question}") for i in range(4): self.options[i].config(text=self.current_options[i], state="normal", bg="SystemButtonFace") self.restore_buttons() def check_answer(self, index): selected = self.options[index]["text"] if selected == self.correct_answer: self.score += 10 messagebox.showinfo("Correct", "That's the correct answer!") else: messagebox.showinfo("Wrong", f"Wrong answer! Correct was: {self.correct_answer}") self.q_index += 1 self.load_question() def use_5050(self): if self.used_5050: messagebox.showinfo("Used", "You have already used 50:50 lifeline.") return self.used_5050 = True correct = self.correct_answer others = [opt for opt in self.current_options if opt != correct] to_remove = random.sample(others, 2) for btn in self.options: if btn["text"] in to_remove: btn.config(state="disabled") def use_poll(self): if self.used_poll: messagebox.showinfo("Used", "You have already used Audience Poll lifeline.") return self.used_poll = True poll_result = [random.randint(10, 30) for _ in range(4)] correct_idx = self.current_options.index(self.correct_answer) poll_result[correct_idx] = max(poll_result) + 20 poll_msg = "\\n".join([f"{self.current_options[i]}: {poll_result[i]}%" for i in range(4)]) messagebox.showinfo("Audience Poll", poll_msg) def use_skip(self): if self.used_skip: messagebox.showinfo("Used", "You have already used Skip Question lifeline.") return self.used_skip = True messagebox.showinfo("Skipped", f"Skipping question! Correct was: {self.correct_answer}") self.q_index += 1 self.load_question() def restore_buttons(self): for btn in self.options: btn.config(state="normal") if __name__ == "__main__": root = tk.Tk() game = KBCGame(root) root.mainloop() """ # Save the code to a .py file code_path = "/mnt/data/kbc_game_gui.py" with open(code_path, "w") as f: f.write(kbc_code) code_path
